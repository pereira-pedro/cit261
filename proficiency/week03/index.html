    Object Literal

    The easiest way to create a JavaScript object is using a collection of properties and values. Each pair (proberty/value) is comma-separated. All object needs to be inside curly braces.

    Example

    var buffer = '';
const student = { name: 'Pedro Pereira', email: 'per17009@byui.edu', city: 'Campo Grande', state: 'MS'}

const student1 = new Student('Pedro Pereira', 'per17009@byui.edu');

    2 - Inheritance
    
        
    
          Modern JavaScript allows classes inheritance. This section shows two
          ways doing class inheritance using JavaScript.
        
    
        
    
          
    Using ES5
    
          
    
            An object must be created much like creating an object using
            Object.create().
          
    
          
    Example
    
          
    var buffer = '';
function Person(name, email) {
    this.name = name;
    this.email = email;
}

Person.prototype.getName = function getName() {
    return this.name;
}

Person.prototype.getEmail = function getEmail() {
    return this.email;
}

function Student(name, email, gpa) {
    this.gpa = gpa;

    Person.call(this, name, email);
}

Student.prototype = Object.create(Person.prototype);
Student.prototype.constructor = Student;
Student.parent = Person.prototype;

Student.prototype.getName = function () {
    return this.name;
}

Student.prototype.getEmail = function () {
    return this.email;
}

Student.prototype.getGPA = function getGPA() {
    return this.gpa;
}

const student = new Student('Pedro Pereira', 'per17009@byui.edu', 3.7);

buffer = `Name: ${student.getName()}\nE-mail: ${student.getEmail()}\nGPA: ${student.getGPA()}`;
    
          
          
    
            
    
          
    
          
    Using ES6
    
          
    
            Using ES6 things are more clear and much alike other
            object-orientaded languages.
          
    
          
    Example
    
          
    var buffer = '';

class Person {
    constructor (name, email) {
        this.name = name;
        this.email = email;
    }

    getName () {
        return this.name;
    }

    getEmail() {
        return this.email;
    }
}

class Student extends Person {
    constructor(name, email, gpa) {

        super(name, email);

        this.gpa = gpa;
    }

    getName() {
        return super.getName();
    }

    getEmail() {
        return super.getEmail();
    }

    getGPA() {
        return this.gpa;
    }
}
const student = new Student('Pedro Pereira', 'per17009@byui.edu', 3.8);

buffer = `Name: ${student.getName()}\nE-mail: ${student.getEmail()}\nGPA: ${student.getGPA()}`;

    3 - Properties
    
        
    
          Properties are data stored by JavaScript objects. In fact, objects in
          JavaScript are a collection of unordered properties.
        
    
        
    
          
    Acessing data
    
          
    
            There are two ways to have access to data.
          
    
          
    Syntax
    
          
    objectName.property      // first
objectName[expression]   // second
              
    
          
    Example
    
          
    var buffer = '';
const student = {name: "Pedro Pereira", email: "per17009@byui.edu"};

buffer += `Name: ${student.name}\nE-mail: ${student.email}\n`;
buffer += `Name: ${student["name"]}\nE-mail: ${student["email"]}`;
    
          
          
    
            
    
          
    
          
    Adding property
    
          
    
            Properties can be added to any JavaScript object.
          
    
          
    Example
    
          
    var buffer = '';
const student = {name: "Pedro Pereira", email: "per17009@byui.edu"};

student.gpa = 3.7;
buffer += `Name: ${student.name}\nE-mail: ${student.email}\nGPA: ${student.gpa}\n`;
    
          
          
    
            
    
          
    
          
    Removing property
    
          
    
            Properties can be removed to any JavaScript object.
          
    
          
    Example
    
          
    var buffer = '';
const student = {name: "Pedro Pereira", email: "per17009@byui.edu", gpa: 3.9};
delete student.gpa;

buffer += `Name: ${student.name}\nE-mail: ${student.email}\nGPA: ${student.hasOwnProperty('gpa')?student.gpa : 'N/A'}\n`;

    5 - Instantiation
    
        
    
          Instantiation are ways to create something in JavasScript. There are
          four methods to create objects. However, all four methods will create
          an object and create methods and properties for that object.
        
    
        
    
          
    Functional
    
          
    
            A function is created, inside it and an empty object is created and
            properties and methods are inserted. Finnaly, the object is
            returned.
          
    
          
    Example
    
          
        var buffer = '';
    const Student = function(name, email) {
        const obj = {};

        obj.name = name;
        obj.email = email;
        obj.gpa = 0;

        obj.setGPA = function(gpa) {
            this.gpa = gpa;
        }

        obj.display = function() {
            return `Name: ${this.name}\nEmail: ${this.email}\nGPA: ${this.gpa}\n`;
        }

        return obj;
    }

    const student = Student("Pedro Pereira", "per17009@byui.edu");
    buffer += student.display();
    student.setGPA(3.6);
    buffer += student.display();
    
          
          
    
            
    
          
    
          
    Functional Shared
    
          
    
            A function is created, inside it and an empty object is created and
            properties inserted. Finnaly, the object is returned. The methods
            are defined in anothe extended object.
          
    
          
    Example
    
          
    var buffer = '';
const Student = function(name, email) {
    const obj = {};

    obj.name = name;
    obj.email = email;
    obj.gpa = 0;

    extend(obj, objMethods);

    return obj;
}
const extend = function(obj, methods) {
    for(var key in methods) {
        obj[key] = methods[key];
    }
}

const objMethods = {
    setGPA: function(gpa) {
        this.gpa = gpa;
    },
    display: function() {
        return `Name: ${this.name}\nEmail: ${this.email}\nGPA: ${this.gpa}\n`;
    }
}

const student = Student("Pedro Pereira", "per17009@byui.edu");
buffer += student.display();
student.setGPA(3.6);
buffer += student.display();
    
          
          
    
            
    
          
    
          
    Prototypal
    
          
    
            Utilizes prototype chain to create objects. The methods are attached
            to object's prototype using Object.create().
          
    
          
    Example
    
          
    var buffer = '';
const Student = function(name, email) {
    const obj = Object.create(objMethods);

    obj.name = name;
    obj.email = email;
    obj.gpa = 0;

    return obj;
}

const objMethods = {
    setGPA: function(gpa) {
        this.gpa = gpa;
    },
    display: function() {
        return `Name: ${this.name}\nEmail: ${this.email}\nGPA: ${this.gpa}\n`;
    }
}

const student = Student("Pedro Pereira", "per17009@byui.edu");
buffer += student.display();
student.setGPA(3.6);
buffer += student.display();

    Example
    
          
    var buffer = '';
const Student = function(name, email) {
    this.name = name;
    this.email = email;
    this.gpa = 0;

Student.prototype.setGPA = function(gpa) {

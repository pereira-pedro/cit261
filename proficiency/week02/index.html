  • 1 - Loops

    Loops are important to execute blocks of code more than one time.

    for

    A for loop executes a block of code a defined number of iterations.

    Syntax

    for (statement 1; statement 2; statement 3) {
  // code block to be executed
}

    Example

    var buffer = '';
for(i=0; i < 5; i++) {
  buffer += `${i} is ${(i % 2) ? 'odd' : 'even'}\n`;
}

    
          
    
          
    while
    
          
    
            A while loop executes a block of code as long as a
            condition evaluates true.
          
    
          
    Syntax
    
          
    while (condition) {
  // code block to be executed
}
        
    
          
    Example
    
          
    var buffer = '';
var i = -1;
while(i++ < 4) {
    buffer += `${i} is ${(i % 2) ? 'odd' : 'even'}\n`;
}

    
          
          
    
            
    
          
    
          
    do
    
          
    
            A do loop executes a block of code as long as a
            condition evaluates true. The main difference is the codition
            evaluation. do executes one time then conditional
            expression is run.
          
    
          
    Syntax
    
          
    do {
  // code block to be executed
}
while (condition);

    
          
    Example
    
          
    var buffer = '';
var i = 0;
do
{
  buffer += `${i} is ${(i % 2) ? 'odd' : 'even'}\n`;
} while(++i < 5)

    2 - Conditional Statements
    
        
    
          
    
            Conditional statements, different blocks of code may be executed
            based in pre-defined conditions.
          
    
          
          
    if
    
          
    
            An if statement executes a block of code if the
            specified condition is true.
          
    
          
    Syntax
    
          
    if (condition) {
  //  block of code to be executed if the condition is true
} else {
  //  block of code to be executed if the condition is false
}
              
    
          
    Example
    

          
    var buffer = '';
const input = document.getElementById('topic-4-input-1').value;

if( input > 5) {
  buffer = "You're almost to reach 10.";
} else {
  buffer = "Keep going. Number 10 is almost there.";
}

    switch
    
          
    
            A switch statement selects one block of code to
            execute.
          
    
          
    Syntax
    
          
    switch(expression) {
  case x:
    // code block
    break;
  case y:
    // code block
    break;
  default:
    // code block
}
              
    
          
    Example
    

          
    var buffer = '';
const input = document.getElementById('topic-5-input-1').value;

switch( parseInt(input) )
{
  case 1:
    buffer = "You're starting your journey to number 10.";
    break;
  case 5:
    buffer = "You're half-way to number 10.";
    break;
  case 9:
    buffer = "One more step to number 10.";
    break;
  case 10:
    buffer = "That's it! You made it!";
    break;
  default:
    buffer = "Keep going.";
}

    5 - Parameters
    
        
    
          
          
    Rules
    
          
    
            
    • JavaScript function definitions do not specify data types for
            parameters.
      • 
            
    • JavaScript functions do not perform type checking on the
            passed arguments.
      • 
            
    • JavaScript functions do not check the number of
            arguments received.
      • 
          
    
          
    
          
    Defaults
    
          
    
            If a function is called with missing arguments (less than declared),
            the missing values are set to: undefined

            Sometimes this is acceptable, but sometimes it is better to assign a
            default value to the parameter.
          
    
          
    Example
    

          
    function sumTwoNumbers(x, y) {
  if (y === undefined) {
    y = 0;
  }

  return x + y;
}
buffer = sumTwoNumbers(10);
    
          
          
    
            
    
          
    
          
    The arguments object
    
          
    
            JavaScript functions have a built-in object called the arguments
            object. The argument object contains an array of the arguments used
            when the function was called (invoked). This way you can simply use
            a function to find (for instance) the highest value in a list of
            numbers.
          
    
          
    Example
    

          
    function findMax() {
  var i;
  var max = -Infinity;
  for (i = 0; i < arguments.length; i++) {
    if (arguments[i] > max) {
      max = arguments[i];
    }
  }
  return max;
}
buffer = findMax(1, 123, 500, 115, 44, 88);

    7 - Associate Arrays
    
            
    
              
              
    
               An array cannot be indexed using strings. Only integers can act as array indexes. However, JavaScript has a mechanism to access a variable associated with an array's object property collection.
               To use an array as associative, a empty array must be create and then elements must be assigned.
              
    
             
              
    Example
    
    
              
    // empty array
var person = []; // empty array

// add elements with string keys
person['name'] = 'Pedro';
person['age'] = 46;
person['city'] = 'Campo Grande';
person['state'] = 'MS';

buffer = `${person['name']}\n${person['age']}\n${person['city']}, ${person['state']}`
